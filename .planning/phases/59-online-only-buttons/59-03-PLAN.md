<interfaces>
Pattern fuer Pages mit Delete-Buttons (presentAlert/ActionSheet):
```typescript
const { isOnline } = useApp();

// Bei presentAlert fuer Delete-Bestaetigungen:
const handleDelete = async (item: Item) => {
  if (!isOnline) return; // Frueh-Abbruch wenn offline
  presentAlert({
    header: 'Loeschen?',
    // ...
  });
};

// Bei IonButton fuer Delete:
<IonButton disabled={!isOnline} onClick={() => handleDelete(item)}>
  {!isOnline ? 'Du bist offline' : 'Loeschen'}
</IonButton>

// Bei IonItemOption (Slide-to-delete):
<IonItemOption disabled={!isOnline} color="danger" onClick={() => handleDelete(item)}>
  Loeschen
</IonItemOption>
```

Pattern fuer inline-Buttons auf Pages:
```typescript
// Einfacher Button:
<IonButton disabled={!isOnline} onClick={handleAction}>
  {!isOnline ? 'Du bist offline' : 'Aktion'}
</IonButton>

// Icon-Button (z.B. Loeschen in Listen):
// Hier KEINEN Text aendern (ist ein Icon), nur disabled setzen
<IonButton disabled={!isOnline} onClick={() => handleDelete(item)}>
  <IonIcon icon={trashOutline} />
</IonButton>
```
