const { Pool } = require('pg');

const TEST_DB_NAME = 'konfi_test';
const ADMIN_URL = process.env.TEST_DATABASE_URL || 'postgresql://postgres:postgres@localhost:5433/postgres';

module.exports = async function globalSetup() {
  // 1. DB droppen falls vorhanden, neu erstellen
  const adminPool = new Pool({ connectionString: ADMIN_URL });
  await adminPool.query(`DROP DATABASE IF EXISTS "${TEST_DB_NAME}"`);
  await adminPool.query(`CREATE DATABASE "${TEST_DB_NAME}"`);
  await adminPool.end();

  // 2. Migrations auf Test-DB ausfuehren
  const testUrl = ADMIN_URL.replace(/\/[^/]+$/, `/${TEST_DB_NAME}`);
  process.env.TEST_DB_URL = testUrl;  // Fuer db.js Helper

  const testPool = new Pool({ connectionString: testUrl });

  // init-scripts zuerst (007_levels.sql etc.)
  const path = require('path');
  const fs = require('fs');
  const initDir = path.join(__dirname, '..', 'init-scripts');
  if (fs.existsSync(initDir)) {
    const initFiles = fs.readdirSync(initDir).filter(f => f.endsWith('.sql')).sort();
    for (const file of initFiles) {
      await testPool.query(fs.readFileSync(path.join(initDir, file), 'utf8'));
    }
  }

  // runMigrations aus database.js wiederverwenden
  // ACHTUNG: database.js hat Top-Level Side-Effect (pool.query + runMigrations).
  // Wir koennen runMigrations NICHT direkt importieren ohne den Side-Effect.
  // Loesung: Die runMigrations-Logik hier duplizieren (15 Zeilen) statt das Modul zu importieren.
  // (database.js refactoring kommt NICHT in Phase 101 — minimal-invasiv)
  const migrationsDir = path.join(__dirname, '..', 'migrations');
  await testPool.query(`CREATE TABLE IF NOT EXISTS schema_migrations (name TEXT PRIMARY KEY, applied_at TIMESTAMPTZ NOT NULL DEFAULT NOW())`);
  const files = fs.readdirSync(migrationsDir).filter(f => f.endsWith('.sql')).sort();
  const { rows: applied } = await testPool.query('SELECT name FROM schema_migrations');
  const appliedSet = new Set(applied.map(r => r.name));
  for (const file of files) {
    if (appliedSet.has(file)) continue;
    const sql = fs.readFileSync(path.join(migrationsDir, file), 'utf8');
    await testPool.query(sql);
    await testPool.query('INSERT INTO schema_migrations (name) VALUES ($1)', [file]);
  }

  await testPool.end();
  console.log(`[globalSetup] Test-DB "${TEST_DB_NAME}" erstellt und migriert (${files.length} Migrations)`);
};
```

const { Pool } = require('pg');

const TEST_DB_NAME = 'konfi_test';
const ADMIN_URL = process.env.TEST_DATABASE_URL || 'postgresql://postgres:postgres@localhost:5433/postgres';
const TEST_DB_URL = ADMIN_URL.replace(/\/[^/]+$/, `/${TEST_DB_NAME}`);

let pool = null;

function getTestPool() {
  if (!pool) {
    pool = new Pool({ connectionString: TEST_DB_URL, max: 5 });
  }
  return {
    query: (text, params) => pool.query(text, params),
    getClient: () => pool.connect(),
    end: () => pool.end(),
  };
}

// TRUNCATE mit CASCADE per D-01, RESTART IDENTITY fuer stabile IDs
async function truncateAll(db) {
  await db.query(`TRUNCATE
    chat_messages, chat_participants, chat_rooms,
    event_bookings, event_timeslots,
    konfi_activities, konfi_badges, bonus_points,
    konfi_profiles, user_jahrgang_assignments,
    users, activities, badges, events,
    jahrgaenge, categories, levels,
    organizations, roles
    RESTART IDENTITY CASCADE`);
}

async function closePool() {
  if (pool) {
    await pool.end();
    pool = null;
  }
}

module.exports = { getTestPool, truncateAll, closePool };
```

**2. backend/tests/helpers/auth.js erstellen (per D-10):**

```javascript
const jwt = require('jsonwebtoken');
const { USERS } = require('./seed');

const JWT_SECRET = process.env.JWT_SECRET || 'test-secret-key-for-vitest';

function generateToken(userKey, expiresIn = '1h') {
  const user = USERS[userKey];
  if (!user) throw new Error(`Unbekannter Seed-User: ${userKey}`);
  return jwt.sign({
    id: user.id,
    type: user.type,
    display_name: user.display_name,
    organization_id: user.org_id,
    role_id: user.role_id,
  }, JWT_SECRET, { expiresIn });
}

function getAllTokens() {
  const tokens = {};
  for (const key of Object.keys(USERS)) {
    tokens[key] = generateToken(key);
  }
  return tokens;
}

module.exports = { generateToken, getAllTokens, SEED_USERS: USERS };
```

KRITISCH per D-10: RBAC wird NIEMALS gemockt. Die Tokens enthalten echte User-IDs die im Seed existieren. rbac.js laedt den User per `decoded.id` aus der DB — der Seed MUSS diese Users enthalten haben.
