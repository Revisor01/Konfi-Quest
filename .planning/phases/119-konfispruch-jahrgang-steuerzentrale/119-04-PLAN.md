<!-- konfspruch-Builder (konfi.js:486-522, 1:1 uebernehmen) -->
Listen-Wahl (konfspruch_id gesetzt): SELECT ks.id, ks.reference, ks.book, ks.chapter, ks.verse, ku.text FROM konfsprueche ks LEFT JOIN konfspruch_uebersetzungen ku ON ku.spruch_id=ks.id AND ku.translation=$translation WHERE ks.id=$konfspruch_id AND ks.is_active=true AND (ks.organization_id IS NULL OR ks.organization_id=$orgId). translation = konfi.konfspruch_translation || 'luther2017'. Ergebnis-Objekt: { source:'liste', id, reference, text:text||'', translation }. Freitext (konfspruch_freitext gesetzt): { source:'freitext', text:konfspruch_freitext, reference:konfspruch_freitext_referenz }. Sonst konfspruch=null.
