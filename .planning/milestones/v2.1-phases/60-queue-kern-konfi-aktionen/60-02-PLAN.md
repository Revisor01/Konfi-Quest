2. sendMessage Funktion (Zeile ~441) erweitern — neuen Offline-Pfad HINZUFUEGEN:

```typescript
const sendMessage = async () => {
  if (!messageText.trim() && !selectedFile) return;
  if (!room) return;

  const clientId = uuidv4();
  const localId = uuidv4();
  const content = messageText.trim();
  const file = selectedFile;

  // Optimistic UI: Nachricht sofort in messages-State einfuegen
  const optimisticMsg: Message = {
    id: -Date.now(), // Temporaere negative ID
    content: content || (file ? file.name : ''),
    sender_id: user?.id ?? 0,
    sender_name: user?.display_name ?? '',
    sender_type: (user?.role_name === 'konfi' ? 'konfi' : 'admin') as 'admin' | 'konfi',
    created_at: new Date().toISOString(),
    message_type: file ? 'image' : 'text',
    file_name: file?.name,
    queueStatus: 'pending',
    localId,
  };

  // UI sofort aktualisieren
  setMessages(prev => [...prev, optimisticMsg]);
  setMessageText('');
  setReplyToMessage(null);
  clearSelectedFile();
  setShouldAutoScroll(true);
  setTimeout(() => contentRef.current?.scrollToBottom(300), 100);

  if (networkMonitor.isOnline) {
    // Online: Normal senden wie bisher
    setUploading(true);
    try {
      const formData = new FormData();
      if (content) formData.append('content', content);
      if (file) formData.append('file', file);
      if (replyToMessage) formData.append('reply_to', replyToMessage.id.toString());
      formData.append('client_id', clientId);

      await api.post(`/chat/rooms/${room.id}/messages`, formData, {
        headers: { 'Content-Type': 'multipart/form-data' }
      });

      // Optimistic msg wird durch WebSocket newMessage Event ersetzt
      // Entferne optimistic msg nach kurzem Delay (WebSocket kommt schnell)
      setTimeout(() => {
        setMessages(prev => prev.filter(m => m.localId !== localId));
      }, 2000);

      if (room) markRoomAsRead();
      setShouldAutoScroll(true);
      await loadMessages();
    } catch (err) {
      // Bei Fehler: optimistic msg als error markieren
      setMessages(prev => prev.map(m =>
        m.localId === localId ? { ...m, queueStatus: 'error' as const } : m
      ));
    } finally {
      setUploading(false);
    }
  } else {
    // Offline: In Queue schreiben (per D-06)
    let hasFileUpload = false;
    let queueBody: any = { content, client_id: clientId };

    if (file) {
      // Bild lokal speichern (per D-08)
      hasFileUpload = true;
      try {
        const reader = new FileReader();
        const base64 = await new Promise<string>((resolve, reject) => {
          reader.onload = () => resolve(reader.result as string);
          reader.onerror = reject;
          reader.readAsDataURL(file);
        });
        // Base64-Daten in Capacitor Filesystem speichern
        const fileName = `queue_${clientId}_${file.name}`;
        await Filesystem.writeFile({
          path: `queue-uploads/${fileName}`,
          data: base64,
          directory: Directory.Data,
        });
        queueBody._localFilePath = `queue-uploads/${fileName}`;
        queueBody._fileName = file.name;
        queueBody._fileType = file.type;
      } catch (fileErr) {
        console.error('Fehler beim lokalen Speichern der Datei:', fileErr);
        setMessages(prev => prev.map(m =>
          m.localId === localId ? { ...m, queueStatus: 'error' as const } : m
        ));
        return;
      }
    }

    if (replyToMessage) queueBody.reply_to = replyToMessage.id.toString();

    await writeQueue.enqueue({
      method: 'POST',
      url: `/chat/rooms/${room.id}/messages`,
      body: queueBody,
      headers: { 'Content-Type': file ? 'multipart/form-data' : 'application/json' },
      maxRetries: 5,
      hasFileUpload,
      metadata: {
        type: 'chat',
        clientId,
        roomId: room.id,
        label: 'Chat-Nachricht',
      },
    });
    // Nachricht bleibt als pending in der UI bis Queue geflusht wird
  }
};
```

WICHTIG zur Queue-Flush-Verarbeitung fuer Chat: Wenn writeQueue.flush() ein Chat-Item mit `_localFilePath` verarbeitet, muss es die Datei aus dem Filesystem lesen und als FormData senden. Dies erfordert eine Erweiterung in writeQueue.ts: Vor dem API-Call pruefen ob `body._localFilePath` existiert, dann Datei lesen, FormData bauen, und nach Erfolg lokale Datei loeschen. ABER: Diese Logik gehoert in writeQueue.ts — in ChatRoom.tsx nur das Enqueue. Falls writeQueue.ts diese FormData-Logik noch nicht hat, sie dort ergaenzen (in der flush-Methode: wenn item.hasFileUpload && item.body._localFilePath, dann Filesystem.readFile + FormData-Konvertierung).

Im "Erneut senden" Handler:
```typescript
handler: async () => {
  // Message als pending markieren
  setMessages(prev => prev.map(m =>
    m.localId === message.localId ? { ...m, queueStatus: 'pending' as const } : m
  ));
  // Queue-Item finden und erneut enqueuen
  const queueItems = await writeQueue.getByMetadata({ roomId: room?.id, type: 'chat' });
  const item = queueItems.find(qi => qi.metadata.clientId === message.localId || qi.id === message.localId);
  if (item) {
    await writeQueue.remove(item.id);
    item.retryCount = 0;
    await writeQueue.enqueue(item);
  }
  // Flush versuchen wenn online
  if (networkMonitor.isOnline) {
    writeQueue.flush();
  }
}
```

Im "Nachricht loeschen" Handler:
```typescript
handler: async () => {
  // Aus UI entfernen
  setMessages(prev => prev.filter(m => m.localId !== message.localId));
  // Aus Queue entfernen
  const queueItems = await writeQueue.getByMetadata({ roomId: room?.id, type: 'chat' });
  const item = queueItems.find(qi => qi.metadata.clientId === message.localId || qi.id === message.localId);
  if (item) {
    await writeQueue.remove(item.id);
    // Lokale Datei loeschen falls vorhanden
    if (item.body?._localFilePath) {
      try {
        await Filesystem.deleteFile({ path: item.body._localFilePath, directory: Directory.Data });
      } catch { /* ignore */ }
    }
  }
}
```
