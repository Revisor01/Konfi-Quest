3. `frontend/src/hooks/useCountUp.ts` erstellen (per D-18, CSS-only per D-22):
   ```typescript
   import { useState, useEffect } from 'react';
   export function useCountUp(target: number, isActive: boolean, duration = 1500): number {
     const [value, setValue] = useState(0);
     useEffect(() => {
       if (!isActive || target <= 0) { setValue(0); return; }
       const start = performance.now();
       let raf: number;
       const tick = (now: number) => {
         const progress = Math.min((now - start) / duration, 1);
         const eased = 1 - Math.pow(1 - progress, 3); // ease-out cubic
         setValue(Math.floor(eased * target));
         if (progress < 1) raf = requestAnimationFrame(tick);
       };
       raf = requestAnimationFrame(tick);
       return () => cancelAnimationFrame(raf);
     }, [target, isActive, duration]);
     return value;
   }
   ```
   Hook startet nur wenn `isActive` true ist (Swiper onSlideChange, per D-21). Ease-out cubic fuer natuerliches Gefuehl. ~1.5s Dauer.
