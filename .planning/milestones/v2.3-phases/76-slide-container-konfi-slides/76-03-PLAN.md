   ```typescript

   import AbschlussSlide from './slides/AbschlussSlide';
   ```

   Slide-Array dynamisch aufbauen (Endspurt ist bedingt):
   ```typescript
   // Slides dynamisch aufbauen
   const slideComponents: React.ReactNode[] = [];
   let slideIndex = 0;

   // Slide 0: Intro
   slideComponents.push(
     <SwiperSlide key="intro">
       <IntroSlide isActive={activeIndex === slideIndex} displayName={displayName} jahrgangName={jahrgangName} year={year!} />
     </SwiperSlide>
   );
   slideIndex++;

   // Slide 1: Punkte
   slideComponents.push(
     <SwiperSlide key="punkte">
       <PunkteSlide isActive={activeIndex === slideIndex} punkte={data.slides.punkte} />
     </SwiperSlide>
   );
   slideIndex++;

   // Slide 2: Events
   slideComponents.push(
     <SwiperSlide key="events">
       <EventsSlide isActive={activeIndex === slideIndex} events={data.slides.events} />
     </SwiperSlide>
   );
   slideIndex++;

   // Slide 3: Badges
   slideComponents.push(
     <SwiperSlide key="badges">
       <BadgesSlide isActive={activeIndex === slideIndex} badges={data.slides.badges} />
     </SwiperSlide>
   );
   slideIndex++;

   // Slide 4: Aktivster Monat
   slideComponents.push(
     <SwiperSlide key="monat">
       <AktivsterMonatSlide isActive={activeIndex === slideIndex} aktivsterMonat={data.slides.aktivster_monat} />
     </SwiperSlide>
   );
   slideIndex++;

   // Slide 5: Chat
   slideComponents.push(
     <SwiperSlide key="chat">
       <ChatSlide isActive={activeIndex === slideIndex} chat={data.slides.chat} />
     </SwiperSlide>
   );
   slideIndex++;

   // Slide 6: Endspurt (bedingt, per D-16 / KS-07)
   if (data.slides.endspurt.aktiv) {
     slideComponents.push(
       <SwiperSlide key="endspurt">
         <EndspurtSlide isActive={activeIndex === slideIndex} endspurt={data.slides.endspurt} />
       </SwiperSlide>
     );
     slideIndex++;
   }

   // Letzter Slide: Abschluss
   slideComponents.push(
     <SwiperSlide key="abschluss">
       <AbschlussSlide isActive={activeIndex === slideIndex} data={data} year={year!} />
     </SwiperSlide>
   );
   ```

   In der Swiper-Komponente: `{slideComponents}` statt fester SwiperSlide-Elemente.
